016package nl.knaw.dans.lib.dataverse;
017
018import com.fasterxml.jackson.core.JsonParser;
019import com.fasterxml.jackson.databind.DeserializationContext;
020import com.fasterxml.jackson.databind.JsonNode;
021import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
022import nl.knaw.dans.lib.dataverse.model.dataset.CompoundField;
023import nl.knaw.dans.lib.dataverse.model.dataset.ControlledMultiValueField;
024import nl.knaw.dans.lib.dataverse.model.dataset.ControlledSingleValueField;
025import nl.knaw.dans.lib.dataverse.model.dataset.MetadataField;
026import nl.knaw.dans.lib.dataverse.model.dataset.PrimitiveMultiValueField;
027import nl.knaw.dans.lib.dataverse.model.dataset.PrimitiveSingleValueField;
028import nl.knaw.dans.lib.dataverse.model.dataset.SingleValueField;
029
030import java.io.IOException;
031import java.util.HashMap;
032import java.util.Iterator;
033import java.util.Map;
034import java.util.stream.Collectors;
035import java.util.stream.StreamSupport;
036
037public class MetadataFieldDeserializer extends StdDeserializer {
038
039    public MetadataFieldDeserializer() {
040        this(null);
041    }
042
043    protected MetadataFieldDeserializer(Class<?> vc) {
044        super(vc);
045    }
046
047    @Override
048    public Object deserialize(JsonParser p, DeserializationContext ctxt) throws IOException {
049        return deserializedMetadataField(p.getCodec().readTree(p), false);
050    }
051
052    private MetadataField deserializedMetadataField(JsonNode n, boolean subField) {
053        String typeName = n.get("typeName").asText();
054        String typeClass = n.get("typeClass").asText();
055        boolean multiple = n.get("multiple").asBoolean();
056        JsonNode valueNode = n.get("value");
057
058        if ("primitive".equals(typeClass)) {
059            if (multiple) {
060                if (subField) throw new IllegalArgumentException("Compound fields cannot contain multi-value subfields");
061                Iterable<JsonNode> jsonNodeIterable = valueNode::elements;
062                return new PrimitiveMultiValueField(
063                    typeClass,
064                    typeName,
065                    true,
066                    StreamSupport.stream(jsonNodeIterable.spliterator(), false)
067                        .map(JsonNode::asText).collect(Collectors.toList()));
068            }
069            else {
070                return new PrimitiveSingleValueField(
071                    typeClass,
072                    typeName,
073                    false,
074                    valueNode.asText());
075            }
076        }
077        else if ("controlledVocabulary".equals(typeClass)) {
078            if (multiple) {
079                if (subField) throw new IllegalArgumentException("Compound fields cannot contain multi-value subfields");
080                Iterable<JsonNode> jsonNodeIterable = valueNode::elements;
081                return new ControlledMultiValueField(
082                    typeClass,
083                    typeName,
084                    true,
085                    StreamSupport.stream(jsonNodeIterable.spliterator(), false)
086                        .map(JsonNode::asText).collect(Collectors.toList()));
087            }
088            else {
089                return new ControlledSingleValueField(
090                    typeClass,
091                    typeName,
092                    false,
093                    valueNode.asText());
094            }
095        }
096        else if ("compound".equals(typeClass)) {
097            if (subField) throw new IllegalArgumentException("Compound fields cannot contain compound fields as subfields");
098            Iterable<JsonNode> jsonNodeIterable = valueNode::elements;
099            return new CompoundField(
100                typeClass,
101                typeName,
102                multiple,
103                StreamSupport.stream(jsonNodeIterable.spliterator(), false)
104                    .map(this::deserializeCompoundFieldValue).collect(Collectors.toList()));
105        }
106
107        return null;
108    }
109
110    private Map<String, SingleValueField> deserializeCompoundFieldValue(JsonNode jsonNode) {
111        Map<String, SingleValueField> subFields = new HashMap<>();
112        for (Iterator<String> it = jsonNode.fieldNames(); it.hasNext(); ) {
113            String subFieldName = it.next();
114            subFields.put(subFieldName, (SingleValueField) deserializedMetadataField(jsonNode.get(subFieldName), true));
115        }
116        return subFields;
117    }
118
119}
