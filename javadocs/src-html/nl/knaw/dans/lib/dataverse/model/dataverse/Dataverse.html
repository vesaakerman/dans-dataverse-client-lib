016package nl.knaw.dans.lib.dataverse.model.dataverse;
017
018import java.util.List;
019
020public class Dataverse {
021    private int id;
022    private int ownerId;
023    private String name;
024    private String alias;
025    private boolean permissionRoot;
026    private String affiliation;
027    private String description;
028    private DataverseType dataverseType;
029    private String storageDriverLabel;
030    private String creationDate;
031    private List<DataverseContact> dataverseContacts;
032
033    public int getId() {
034        return id;
035    }
036
037    public void setId(int id) {
038        this.id = id;
039    }
040
041    public int getOwnerId() {
042        return ownerId;
043    }
044
045    public void setOwnerId(int ownerId) {
046        this.ownerId = ownerId;
047    }
048
049    public String getName() {
050        return name;
051    }
052
053    public void setName(String name) {
054        this.name = name;
055    }
056
057    public String getAlias() {
058        return alias;
059    }
060
061    public void setAlias(String alias) {
062        this.alias = alias;
063    }
064
065    public boolean isPermissionRoot() {
066        return permissionRoot;
067    }
068
069    public void setPermissionRoot(boolean permissionRoot) {
070        this.permissionRoot = permissionRoot;
071    }
072
073    public String getAffiliation() {
074        return affiliation;
075    }
076
077    public void setAffiliation(String affiliation) {
078        this.affiliation = affiliation;
079    }
080
081    public String getDescription() {
082        return description;
083    }
084
085    public void setDescription(String description) {
086        this.description = description;
087    }
088
089    public DataverseType getDataverseType() {
090        return dataverseType;
091    }
092
093    public void setDataverseType(DataverseType dataverseType) {
094        this.dataverseType = dataverseType;
095    }
096
097    public String getStorageDriverLabel() {
098        return storageDriverLabel;
099    }
100
101    public void setStorageDriverLabel(String storageDriverLabel) {
102        this.storageDriverLabel = storageDriverLabel;
103    }
104
105    public String getCreationDate() {
106        return creationDate;
107    }
108
109    public void setCreationDate(String creationDate) {
110        this.creationDate = creationDate;
111    }
112
113    public List<DataverseContact> getDataverseContacts() {
114        return dataverseContacts;
115    }
116
117    public void setDataverseContacts(List<DataverseContact> dataverseContacts) {
118        this.dataverseContacts = dataverseContacts;
119    }
120}
