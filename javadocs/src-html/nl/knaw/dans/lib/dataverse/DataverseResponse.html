016package nl.knaw.dans.lib.dataverse;
017
018import com.fasterxml.jackson.databind.JavaType;
019import com.fasterxml.jackson.databind.JsonNode;
020import com.fasterxml.jackson.databind.ObjectMapper;
021import com.fasterxml.jackson.databind.type.TypeFactory;
022import nl.knaw.dans.lib.dataverse.model.DataverseEnvelope;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026import java.io.IOException;
027
028/**
029 * Response from Dataverse. A typical response from Dataverse is a JSON document with the following format:
030 *
031 * <!-- @formatter:off -->
032 * ```json
033 *  {
034 *    "status": "OK",
035 *    "data": {
036 *        "myfield": "myvalue"
037 *    }
038 *  }
039 * ```
040 * <!-- @formatter:on -->
041 *
042 * {@link nl.knaw.dans.lib.dataverse.model}.
043 *
044 * @param <D> the type of the data of the response message envelope
045 */
046public class DataverseResponse<D> {
047    private static final Logger log = LoggerFactory.getLogger(DataverseResponse.class);
048    private final ObjectMapper mapper;
049
050    private final String bodyText;
051    private final JavaType dataType;
052
053    protected DataverseResponse(String bodyText, ObjectMapper mapper, Class<?>... dataClass) {
054        log.trace("ENTER");
055        log.trace(bodyText);
056        this.bodyText = bodyText;
057        this.mapper = mapper;
058        TypeFactory typeFactory = mapper.getTypeFactory();
059        if (dataClass.length == 2) {
060            JavaType inner = typeFactory.constructParametricType(dataClass[0], dataClass[1]);
061            this.dataType = typeFactory.constructParametricType(DataverseEnvelope.class, inner);
062        }
063        else {
064            this.dataType = typeFactory.constructParametricType(DataverseEnvelope.class, dataClass[0]);
065        }
066    }
067
068    // Must be static, otherwise compiler complains about passing result into 'this()' call
069    private static JavaType constuctJavaTypeForContainerClass(Class<?> containerClass, Class<?> elementClass, ObjectMapper mapper) {
070        TypeFactory typeFactory = mapper.getTypeFactory();
071        JavaType inner = typeFactory.constructParametricType(containerClass, elementClass);
072        return typeFactory.constructParametricType(DataverseEnvelope.class, inner);
073    }
074
075    /**
076     * @return A dataverse envelope
077     * @throws IOException if body cannot be processed properly as JSON
078     */
079    public DataverseEnvelope<D> getEnvelope() throws IOException {
080        return mapper.readValue(bodyText, dataType);
081    }
082
083    /**
084     * @return the payload of the envelope directly
085     * @throws IOException if body cannot be processed properly as JSON
086     */
087    public D getData() throws IOException {
088        return getEnvelope().getData();
089    }
090
091    /**
092     * @return the envelope as a JSON AST
093     * @throws IOException if body cannot be processed properly as JSON
094     */
095    public JsonNode getEnvelopeAsJson() throws IOException {
096        return mapper.readTree(bodyText);
097    }
098
099    /**
100     * @return the body as a String
101     */
102    public String getEnvelopeAsString() {
103        return bodyText;
104    }
105}
