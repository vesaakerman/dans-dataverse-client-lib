043    private static final Logger log = LoggerFactory.getLogger(DataverseApi.class);
044    private final Path subPath;
045
046    protected DataverseApi(HttpClientWrapper httpClientWrapper, String alias) {
047        super(httpClientWrapper);
048        log.trace("ENTER");
049        this.subPath = Paths.get("api/dataverses/").resolve(alias + "/");
050    }

052    /**
053     * See [Dataverse API Guide].
054     *
055     * [Dataverse API Guide]: https://guides.dataverse.org/en/latest/api/native-api.html#create-a-dataverse-collection
056     *
057     * @param dataverse dataverse to create
058     * @return description of the dataverse just created
059     * @throws IOException        when I/O problems occur during the interaction with Dataverse
060     * @throws DataverseException when Dataverse fails to perform the request
061     */
062    public DataverseHttpResponse<Dataverse> create(String dataverse) throws IOException, DataverseException {
063        log.trace("ENTER");
064        return httpClientWrapper.postJsonString(subPath, dataverse, new HashMap<>(), new HashMap<>(), Dataverse.class);
065    }
