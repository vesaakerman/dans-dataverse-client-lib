016package nl.knaw.dans.lib.dataverse;
017
018import com.fasterxml.jackson.core.JsonParser;
019import com.fasterxml.jackson.core.JsonProcessingException;
020import com.fasterxml.jackson.databind.DeserializationContext;
021import com.fasterxml.jackson.databind.JsonNode;
022import com.fasterxml.jackson.databind.deser.std.StdDeserializer;
023import nl.knaw.dans.lib.dataverse.model.dataverse.DataverseDatasetItem;
024import nl.knaw.dans.lib.dataverse.model.dataverse.DataverseItemType;
025import nl.knaw.dans.lib.dataverse.model.dataverse.DataverseSubverseItem;
026
027import java.io.IOException;
028import java.net.URI;
029import java.net.URISyntaxException;
030import java.util.Optional;
031
032public class DataverseItemDeserializer extends StdDeserializer {
033
034    public DataverseItemDeserializer() {
035        this(null);
036    }
037
038    public DataverseItemDeserializer(Class vc) {
039        super(vc);
040    }
041
042    @Override
043    public Object deserialize(JsonParser p, DeserializationContext ctxt) throws IOException, JsonProcessingException {
044        JsonNode node = p.getCodec().readTree(p);
045        String dataverseItemType = node.get("type").asText();
046        if (DataverseItemType.dataverse.toString().equals(dataverseItemType)) {
047            return new DataverseSubverseItem(node.get("title").asText(), node.get("id").asInt());
048        }
049        else if (DataverseItemType.dataset.toString().equals(dataverseItemType)) {
050            DataverseDatasetItem item = new DataverseDatasetItem();
051            Optional.ofNullable(node.get("id")).ifPresent(n -> item.setId(n.asInt()));
052            Optional.ofNullable(node.get("persistentUrl")).ifPresent(n -> item.setPersistentUrl(createUri(n.asText())));
053            Optional.ofNullable(node.get("protocol")).ifPresent(n -> item.setProtocol(n.asText()));
054            Optional.ofNullable(node.get("authority")).ifPresent(n -> item.setAuthority(n.asText()));
055            Optional.ofNullable(node.get("publisher")).ifPresent(n -> item.setPublisher(n.asText()));
056            Optional.ofNullable(node.get("publicationDate")).ifPresent(n -> item.setPublicationDate(n.asText()));
057            Optional.ofNullable(node.get("storageIdentifier")).ifPresent(n -> item.setStorageIdentifier(n.asText()));
058            Optional.ofNullable(node.get("metadataLanguage")).ifPresent(n -> item.setMetadataLanguage(n.asText()));
059            return item;
060        }
061        else {
062            throw new IllegalArgumentException("Not a valid dataverse item type: " + dataverseItemType);
063        }
064    }
065
066    private URI createUri(String s) {
067        try {
068            return new URI(s);
069        }
070        catch (URISyntaxException e) {
071            throw new IllegalArgumentException("Not a valid URI: " + s);
072        }
073    }
074}
